559/// The backlights use 16 dmx channels each. The usage of each channel is as described by the DMX chart found at the manufacture website:
560/// https://www.robe.cz/ledbeam-150/download/#dmx-charts (12/04/2020).
561/// The backlights are in 16 channel mode (mode 2).
562/// Channel : Usage (explaination)
563/// 1: Pan (positioning)
564/// 2: Pan Fine (positioning)
565/// 3: Tilt (positioning)
566/// 4: Tilt Fine (positioning)
567/// 5: Pan/Tilt Speed (positioning)
568/// 6: Power/Special Functions (Control channel)
569/// 7: Virtual Colour Wheel (colour)
570/// 8: Red
571/// 9: Green
572/// 10:Blue
573/// 11:White
574/// 12: CTC (colour temperature)
575/// 13: Colour-Mix-Control
576/// 14: Zoom (beam wideness)
577/// 15: Shutter/Strobe (should the fixture rapidly flash)
578/// 16: Dimmer Intensity (brightness)
579///
580/// This is why this demo is locked to this specific lighting fixture and mode. There is no standard to these channel orderings and so any different
581/// fixture or mode wouldn't behave as expected.
582///
5830fn gen_acceptance_test_step_1_backlight_state(buf: &mut [u8]) {
584 // Position the fixtures pointing at the stage.
5850 buf[0] = 128; // Pan
5860 buf[1] = 0; // Pan Fine
5870 buf[2] = 148; // Tilt
5880 buf[3] = 114; // Tilt Fine
5890 buf[4] = 0; // Pan/Tilt Speed
590
591 // Leave the fixtures in their default options state.
5920 buf[5] = 0; // Power/Special Functions (Control channel)
593
594 // Set the fixture to white.
5950 buf[6] = 0; // Using Red-Green-Blue-White mixing so don't use the virtual colour wheel.
5960 buf[7] = 0; // Red at 0.
5970 buf[8] = 0; // Green at 0.
5980 buf[9] = 0; // Blue at 0.
5990 buf[10] = 255; // White at full.
6000 buf[11] = 0; // Colour temperature at 0 (meaning default colour temperature of around 6500k).
6010 buf[12] = 45; // Use additive colour mixing. 45 is the default value.
602
603 // Set the fixture so it covers the stage.
6040 buf[13] = 91; // Zoom the fixture so it is wide enough to get reasonable coverage.
605
606 // Set the fixture to full.
6070 buf[14] = 255; // The shutter is set to 255 to indicate it is fully open meaning all light can pass.
6080 buf[15] = 255; // The brightness is set to 255 to indicate it should be at full.
6090}

704/// Returns Ok(true) to continue or Ok(false) if no more input.
7050fn handle_input(src: &mut SacnSource) -> Result <bool>{
7060 let mut input = String::new();
707
7080 match io::stdin().read_line(&mut input) {
7090 Ok(n) => {
7100 if n == 0 {
711 // Means EOF is reached so terminate
7120 return Ok(false)
713 }
714
715 // https://www.tutorialspoint.com/rust/rust_string.htm (03/02/2020)
7160 let split_input: Vec<&str> = input.split_whitespace().collect();
7170 if split_input.len() < 2 {
7180 display_help();
7190 bail!(std::io::Error::new(std::io::ErrorKind::InvalidInput, "Insufficient parts ( < 2 )"));
720 }
721
7220 match split_input[0] {
7230 ACTION_IGNORE => {
724 // Ignore Input
7250 Ok(true)
726 }
7270 ACTION_DATA_OPTION => {
7280 handle_data_option(src, split_input)
729 }
7300 ACTION_FULL_DATA_OPTION => {
7310 handle_full_data_option(src, split_input)
732 }
7330 ACTION_UNICAST_OPTION => {
7340 handle_unicast_option(src, split_input)
735 }
7360 ACTION_DATA_OVER_TIME_OPTION => {
7370 handle_data_over_time_option(src, split_input)
738 }
7390 ACTION_SYNC_OPTION => {
7400 let universe: u16 = split_input[1].parse().unwrap();
7410 src.send_sync_packet(universe, None)?;
7420 Ok(true)
743 }
7440 ACTION_UNICAST_SYNC_OPTION => {
7450 if split_input.len() < 3 {
7460 bail!(std::io::Error::new(std::io::ErrorKind::InvalidInput, "Insufficient parts for data line ( < 3 )"));
747 }
748
7490 let universe: u16 = split_input[1].parse().unwrap();
7500 let dst_ip = split_input[2];
7510 src.send_sync_packet(universe, Some(SocketAddr::from_str(dst_ip).unwrap().into()))?;
7520 Ok(true)
753 }
7540 ACTION_REGISTER_OPTION => {
7550 let universe: u16 = split_input[1].parse().unwrap();
7560 src.register_universe(universe)?;
7570 Ok(true)
758 }
7590 ACTION_PREVIEW_OPTION => {
7600 let val = split_input[1].parse();
761
7620 match val {
7630 Ok(v) => {
7640 src.set_preview_mode(v)?;
765 },
766 Err(_e) => {
7670 bail!(std::io::Error::new(std::io::ErrorKind::InvalidInput, "Preview flag option not 'true'/'false' or otherwise parsable as boolean"));
768 }
769 }
7700 Ok(true)
771 }
7720 ACTION_TERMINATE_OPTION => {
7730 let universe: u16 = split_input[1].parse().unwrap();
7740 if universe == 0 {
7750 return Ok(false)
776 } else {
7770 src.terminate_stream(universe, TERMINATE_START_CODE)?;
778 }
7790 Ok(true)
780 }
7810 ACTION_SLEEP_OPTION => {
7820 if split_input.len() < 2 {
7830 display_help();
7840 bail!(std::io::Error::new(std::io::ErrorKind::InvalidInput, "Insufficient parts ( < 2 )"));
785 }
7860 let millis: u64 = split_input[1].parse().unwrap();
7870 sleep(Duration::from_millis(millis));
7880 Ok(true)
789 }
7900 ACTION_ALL_DATA_OPTION => {
7910 handle_all_data_option(src, split_input)
792 }
7930 ACTION_TEST_PRESENT_OPTION => {
7940 handle_test_preset_option(src, split_input)
795 }
7960 x => {
7970 bail!(std::io::Error::new(std::io::ErrorKind::InvalidInput, format!("Unknown input type: {}", x)));
798 }
799 }
8000 }
8010 Err(e) => {
8020 bail!(e);
803 }
804 }
