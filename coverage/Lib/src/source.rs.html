59/// A DMX over sACN sender.
60///
61/// SacnSourceInternal is used for sending sACN packets over ethernet.
62///
63/// # Examples
64///
65/// ```
66/// // Example showing creation of a source and then sending some data.
67/// use sacn::source::SacnSource;
68/// use sacn::packet::ACN_SDT_MULTICAST_PORT;
69/// use std::net::{IpAddr, SocketAddr};
70///
71/// let local_addr: SocketAddr = SocketAddr::new(IpAddr::V4("0.0.0.0".parse().unwrap()), ACN_SDT_MULTICAST_PORT + 1);
72///
73/// let mut src = SacnSource::with_ip("Source", local_addr).unwrap();
74///
75/// let universe: u16 = 1; // Universe the data is to be sent on.
76/// let sync_uni: Option<u16> = None; // Don't want the packet to be delayed on the receiver awaiting synchronisation.
77/// let priority: u8 = 100; // The priority for the sending data, must be 1-200 inclusive, None means use default.
78/// let dst_ip: Option<SocketAddr> = None; // Sending the data using IP multicast so don't have a destination IP.
79///
80/// src.register_universe(universe).unwrap(); // Register with the source that will be sending on the given universe.
81///
82/// let mut data: Vec<u8> = vec![0, 0, 0, 0, 255, 255, 128, 128]; // Some arbitrary data, must have length <= 513 (including start-code).
83///
84/// src.send(&[universe], &data, Some(priority), dst_ip, sync_uni).unwrap(); // Actually send the data
85/// ```
86///
87/// An ANSI E1.31-2018 sACN source.
88///
89/// Allows sending DMX data over an IPv4 or IPv6 network using sACN.
